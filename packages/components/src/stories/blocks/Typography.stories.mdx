import { Meta, Typeset } from '@storybook/addon-docs/blocks'
import { FontFamilies, FontSizes, FontWeights } from 'storybook/config'

<Meta title='Blocks/Typography' />

export const getTypset = family => {
  return Object.values(FontWeights).map(weight => {
    return (
      <Typeset
        fontFamily={family}
        fontSizes={Object.values(FontSizes)}
        fontWeight={weight}
      />
    )
  })
}
