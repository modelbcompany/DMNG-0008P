import { Meta, Typeset } from '@storybook/addon-docs/blocks'
import { FontFamilies, FontSizes, FontWeights } from 'storybook/config'

<Meta title='Blocks/Typography' />

export const getTypset = family => {
  return Object.values(FontWeights).map(weight => {
    return (
      <Typeset
        fontFamily={family}
        fontSizes={Object.values(FontSizes)}
        fontWeight={weight}
      />
    )
  })
}

# Typography

Families: **IBM Plex Mono, Swiss_721 W01**

<br />

## Usage

<br />

### HTML

```html
<!-- IBM Plex Mono -->
<link
  type="text/css"
  rel="stylesheet"
  href="//fonts.googleapis.com/css2?family=IBM+Plex+Mono:wght@300;400;500;600;700&display=swap"
/>

<!-- Swiss 721 -->
<link
  type="text/css"
  rel="stylesheet"
  href="//fast.fonts.net/cssapi/ea5e2124-bbdb-4e91-8a98-9b549b432fdb.css"
/>
```

### Sass

```scss
$font-family-primary: IBM Plex Mono, monospace;
$font-family-secondary: Swiss_721 W01, sans-serif;
```
